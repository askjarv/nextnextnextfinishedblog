Recently I put together something others might find help if you've got a large amount of CCM assets to license and set up with a Compliance Scan Profile... CCM's API isn't super well documented, and the associations between scantasks, assets, etc isn't immediately obvious so hopefully this example (which includes some extra functions that might also be helpful to you all in other use cases) will be of help:

```

Set-CCMScanHeaders -Username $CCMUsername -Password $CCMPassword -Server $CCMServer    $payload = @{licensed=$true} | ConvertTo-Json    Write-host $payload    try{        Invoke-WebRequest -Method "PATCH" -ContentType "application/json" -Uri "https://$ccmserver/api/v1/asset/$AssetID" -Headers $headers -Body $payload        }    Catch        {Write-Error "Failed to license asset"}}function "/api/v1/assetgroup/1";}} | ConvertTo-Json    # Write-host $Payload -ForegroundColor Yellow    Write-debug $Headers    try{        $r = Invoke-RestMethod -UseBasicParsing -Uri "https://$CCMServer/api/v1/asset/$assetid/scantasks" -Headers $headers -Method Post -ContentType text/json -body $payload        return $r        }    catch{      $Error[0]    Write-Error "Failed to add scan profile"    }  }function Invoke-CCMScanTask{    <#  .SYNOPSIS    Executes a scan task immediately  .DESCRIPTION    Executes a scan task immediately  .EXAMPLE     Invoke-CCMScanTask -ID 1
